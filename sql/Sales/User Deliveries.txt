SELECT      T0.[DocEntry] AS 'Belge Numarası',     T0.[CreateDate] AS 'Oluşturma Tarihi',    YEAR(T0.[CreateDate]) AS 'Yıl',        MONTH(T0.[CreateDate]) AS 'Ay',       T0.[UpdateDate] AS 'Güncelleme Tarihi',     T1.[WhsCode] AS 'Ambar Kodu',     T2.[U_NAME] AS 'Kullanıcı Adı',      T1.ItemCode,     T1.Dscription,     T0.[CANCELED] AS 'İptal Durumu' FROM      ODLN T0 INNER JOIN      DLN1 T1 ON T0.[DocEntry] = T1.[DocEntry] INNER JOIN      OUSR T2 ON T0.[UserSign] = T2.[USERID] ORDER BY  T0.[CreateDate] desc
