T0.Quantity > 0 THEN RDR.Quantity - T0.Quantity         ELSE 0     END AS 'Sevk Edilemeyen Miktar',     T0.[UomCode] AS 'Ölçü birimi kodu',     T0.[PriceBefDi] AS 'Birim fiyat',     T1.[DocCur] AS 'Belge para birimi',      T0.[Rate] AS 'Para birimi kuru',      CASE          WHEN T0.[Rate] = 0 OR T0.[Rate] IS NULL THEN T0.[PriceBefDi]         ELSE T0.[PriceBefDi] * T0.[Rate]     END AS 'TRY Karşılığı Birim Fiyat',      T0.[Quantity] *      CASE          WHEN T0.[Rate] = 0 OR T0.[Rate] IS NULL THEN T0.[PriceBefDi]         ELSE T0.[PriceBefDi] * T0.[Rate]     END AS 'Toplam TRY Satış (Vergi Hariç)',      T0.[VatGroup],      CASE          WHEN T0.[VatGroup] = 's05' THEN ' %0'         WHEN T0.[VatGroup] = 's19' THEN ' %20'         WHEN T0.[VatGroup] = 's20' THEN ' %10'         WHEN T0.[VatGroup] = 's21' THEN ' %20'         WHEN T0.[VatGroup] = 'S04' THEN '%0'         WHEN T0.[VatGroup] = 'S12' THEN '%18'         WHEN T0.[VatGroup] = 'S22' THEN '%10'         WHEN T0.[VatGroup] = 'S23' THEN '%20'         WHEN T0.[VatGroup] = 'S24' THEN '%10'         ELSE ''     END AS 'Vergi oranı' FROM       DLN1 T0   INNER JOIN      ODLN T1 ON T1.DocEntry = T0.DocEntry INNER JOIN      OCRD C ON T1.CardCode = C.CardCode INNER JOIN      OCRG G ON C.GroupCode = G.GroupCode INNER JOIN      OSLP SLP ON T1.SlpCode = SLP.SlpCode INNER JOIN      OITM ITM ON T0.ItemCode = ITM.ItemCode INNER JOIN      OITB ITB ON ITM.ItmsGrpCod = ITB.ItmsGrpCod LEFT JOIN      RDR1 RDR ON T0.BaseEntry = RDR.DocEntry AND T0.BaseLine = RDR.LineNum AND T0.BaseType = 17 WHERE      T1.CANCELED = 'N'
