'Müşteri/Satıcı Grup Kodu',     G.[GroupName] AS 'Müşteri/Satıcı Grup Adı',     CASE          WHEN T1.[ShipToCode] <> '' THEN T1.[ShipToCode]         ELSE 'FATURA ADRESİNE SEVK'     END AS 'Sevkiyat T1.[SlpCode] AS 'Satış çalışanı kodu',      SLP.[SlpName] AS 'Satış çalışanı adı',      T0.[WhsCode] AS 'Depo kodu',     ITM.[ItmsGrpCod] AS 'Kalem grubu kodu',     ITB.[ItmsGrpNam] AS 'Kalem grubu adı',     ITM.[U_stokkodgrup] AS 'Stok Kod Grubu',     T0.[ItemCode] AS 'Kalem numarası',      T0.[Dscription] AS 'Kalem/hizmet tanımı',      T0.[Quantity] AS 'Miktar',      T0.[DelivrdQty] AS 'Teslim T0.[Rate] IS NULL THEN T0.[PriceBefDi]         ELSE T0.[PriceBefDi] * T0.[Rate]     END AS 'Teslim Miktarı Toplam TRY Satış (Vergi Hariç)',      T0.[VatGroup],      CASE          WHEN T0.[VatGroup] = 's05' THEN ' %0'         WHEN T0.[VatGroup] = 's19' THEN ' %20'         WHEN T0.[VatGroup] = 's20' THEN ' %10'         WHEN T0.[VatGroup] = 's21' THEN ' %20'         WHEN T0.[VatGroup] = 'S04' THEN '%0'         WHEN T0.[VatGroup] = 'S12' THEN '%18'         WHEN T0.[VatGroup] = 'S22' THEN '%10'         WHEN T0.[VatGroup] = 'S23' THEN '%20'         WHEN T0.[VatGroup] = 'S24' THEN '%10'         ELSE ''     END AS 'Vergi oranı' FROM       [dbo].[RDR1] T0   INNER JOIN      [dbo].[ORDR] T1 ON T1.[DocEntry] = T0.[DocEntry] INNER JOIN      [dbo].[OCRD] C ON T1.[CardCode] = C.[CardCode] INNER JOIN [dbo].[OITB] ITB ON ITM.[ItmsGrpCod] = ITB.[ItmsGrpCod] INNER JOIN      [dbo].[OCTG] OCTG ON T1.[GroupNum] = OCTG.[GroupNum] WHERE      T1.[CANCELED] = 'N' ORDER BY      T1.[DocEntry]
