SELECT      T0."DocNum" AS "Satış Siparişi No",      T0."DocDate" AS "Sipariş Kayıt Tarihi",     T1.LineStatus,      T1."ItemCode" AS "Malzeme Kodu",      T1."Dscription" AS "Malzeme Açıklaması",      T1."Quantity" AS "Sipariş Miktarı",      T1."DelivrdQty" AS "Teslim Edilen Miktar",      T1."OpenQty" AS "Sevk Edilmeyen Miktar",     T1."ShipDate" AS "Sipariş Teslim Tarihi",     DATEDIFF(DAY, T1."ShipDate", GETDATE()) AS "Gecikme Günü",     CASE          WHEN T1."OpenQty" > 0 AND GETDATE() > T1."ShipDate" THEN 'Gecikti'         ELSE 'Zamanında'     END AS "Teslim Durumu",     T3."DocNum" AS "Teslimat Belge No",     T3."DocDate" AS "Teslimat Kayıt Tarihi" FROM      "ORDR" T0 -- Satış Siparişi Başlık Tablosu INNER JOIN      "RDR1" T1 -- Satış Siparişi Kalem Tablosu ON      T0."DocEntry" = T1."DocEntry" LEFT JOIN      "DLN1" T2 -- Teslimat Belgesi Kalem Tablosu ON      T1."DocEntry" = T2."BaseEntry" AND      T1."LineNum" = T2."BaseLine" LEFT JOIN      "ODLN" T3 -- Teslimat Belgesi Başlık Tablosu ON      T2."DocEntry" = T3."DocEntry" WHERE      T0."DocStatus" = 'O'  ORDER BY     T0."DocNum" DESC,     T1."LineNum" ASC
