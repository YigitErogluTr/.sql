SELECT      T1.ItemCode AS [Ürün Kodu],     T2.ItemName AS [Ürün Adı],     SUM(T1.Quantity) AS [Satış Miktarı],     SUM(T1.LineTotal) AS [Toplam Satış Tutarı] FROM      OINV T0 INNER JOIN      INV1 T1 ON T0.DocEntry = T1.DocEntry INNER JOIN      OITM T2 ON T1.ItemCode = T2.ItemCode WHERE      T0.DocDate BETWEEN [%0] AND [%1] GROUP BY      T1.ItemCode, T2.ItemName ORDER BY      [Satış Miktarı] DESC
