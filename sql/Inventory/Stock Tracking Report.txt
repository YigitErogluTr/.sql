SELECT      'EGEMEN' AS [Şirket],       T0.[DocDate] as 'Belge Tarihi',      T0.[TransType] as 'Belge Türü Kodu ',      CASE T0.[TransType]         WHEN 20 THEN 'SAS MAl Girişi'         WHEN 59 THEN 'Mal Giriş'         WHEN 60 THEN 'Mal Çıkışı'         WHEN 67 THEN 'Stok Nakli'         WHEN 15 THEN 'Teslimat'         WHEN 21 THEN 'Mal İadesi'         WHEN 69 THEN 'Mal Girişi'         ELSE 'Diğer / Tanımsız'     END AS [Belge Türü Adı],     T0.[BASE_REF] as 'Belge No',      T0.[ItemCode],      T0.[Dscription],      T1.[InvntryUom] AS [Ölçü Birimi],     T0.[InQty],     T0.[OutQty],      T0.[Warehouse],      U.[USER_CODE] AS [Kullanıcı Adı],     T0.[Comments],      T0.[JrnlMemo]  FROM OINM T0  LEFT JOIN OITM T1 ON T0.[ItemCode] = T1.[ItemCode] LEFT JOIN OUSR U ON T0.[UserSign] = U.[USERID] WHERE T0.[Warehouse] IN ('K.PASA', 'ISKKNT')   AND T0.[DocDate] >= CAST(DATEADD(day, -1, CAST(GETDATE() AS date)) AS datetime)   AND T0.[DocDate] < CAST(CAST(GETDATE() AS date) AS datetime)
