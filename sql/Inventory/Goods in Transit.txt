SELECT T1.[DocNum] ,T1.[DocStatus] AS 'Belge durumu', T1.[DocDueDate] AS 'Planlanan Mal Kabul Tarihi', T0.[ItemCode] AS 'Kalem numarası', T0.[Dscription] AS 'Kalem/hizmet tanımı', T0.[Quantity] AS 'Miktar', T0.[Price] AS 'Birim Fiyat', T0.[Currency] AS 'Fiyat Para Birimi', T0.[LineTotal] AS 'TL SATIR TOPLAMI', T0.[Rate] AS 'Para birimi kuru', T0.[WhsCode] AS 'Depo kodu', T0.[UomCode] FROM  [dbo].[POR1] T0  INNER  JOIN [dbo].[OPOR] T1  ON  T1.[DocEntry] = T0.[DocEntry]   WHERE T1.[DocStatus] = ('O' )
