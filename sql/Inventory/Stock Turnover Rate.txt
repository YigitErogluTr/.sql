SELECT      T1.[ItemCode],       T1.[Dscription],     SUM(T1.[Quantity]) AS "Satış Miktarı(Fatura)",      ISNULL(AVG(T2.OnHand), 0) AS "Stok",       ISNULL(SUM(T1.[Quantity]) / NULLIF(AVG(T2.OnHand), 0), 0) AS "Stok Devir Orani",     DATEDIFF(DAY, MIN(T0.DocDate), MAX(T0.DocDate)) AS "Kalem HareketGunu",       ISNULL((ISNULL(SUM(T1.[Quantity]) / NULLIF(AVG(T2.OnHand), 0), 0)) / NULLIF(DATEDIFF(DAY, MIN(T0.DocDate), MAX(T0.DocDate)), 0), 0) AS "Stok Devir Hızı",    MIN(T0.DocDate) AS "İlk Fatura Tarihi",    MAX(T0.DocDate) AS "Son Fatura Tarihi" FROM      OINV T0   INNER JOIN      INV1 T1 ON T0.DocEntry = T1.DocEntry   LEFT JOIN      OITM T2 ON T1.ItemCode = T2.ItemCode   GROUP BY      T1.ItemCode, T1.[Dscription] ORDER BY    "Stok Devir Hızı" DESC
