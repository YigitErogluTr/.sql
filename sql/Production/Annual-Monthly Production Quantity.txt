SELECT     YEAR(T0.[PostDate]) AS 'Yıl',          MONTH(T0.[PostDate]) AS 'Ay',     T0.[ItemCode],      T0.[ProdName],     COUNT(T0.[DocNum]) AS ToplamBelge,      SUM(T0.[PlannedQty]) AS 'Toplam Planlanan Miktar',      SUM(T0.[CmpltQty]) AS 'Toplam Tamamlanan Miktar',      OITB.ItmsGrpNam AS 'Grup Adı',      OITM.U_stokkodgrup AS 'Stok Kodu Grubu',      OPRC.PrcName AS 'Masraf Yeri Adı',     OITM.U_urungrup AS 'Ürün Grubu' FROM      OWOR T0   LEFT JOIN      OITM ON T0.ItemCode = OITM.ItemCode  LEFT JOIN      OITB ON OITM.ItmsGrpCod = OITB.ItmsGrpCod LEFT JOIN      OPRC ON OITM.U_H1_UretimBolumu = OPRC.PrcCode WHERE  T0.[PostDate] BETWEEN [%0] AND [%1] GROUP BY      YEAR(T0.[PostDate]),      MONTH(T0.[PostDate]),      T0.[ItemCode],      T0.[ProdName],     OITB.ItmsGrpNam,      OITM.U_stokkodgrup,      OPRC.PrcName,      OITM.U_urungrup ORDER BY      Yıl DESC,      Ay DESC,      T0.[ItemCode]
