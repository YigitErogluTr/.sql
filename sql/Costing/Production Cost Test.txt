SELECT     YEAR(OWOR.PostDate) AS [Yıl],     MONTH(OWOR.PostDate) AS [Ay],     OWOR.PostDate AS [Belge Tarihi],     OWOR.DocNum AS [İş Emri No],     OWOR.ItemCode,     OWOR.ProdName,     OWOR.CmpltQty AS [Gerçekleşen Üretim Miktarı],     ISNULL(Maliyet.ToplamMaliyet, 0) AS [Toplam Üretim Maliyeti],     CASE          WHEN OWOR.CmpltQty > 0 THEN ISNULL(Maliyet.ToplamMaliyet, 0) / OWOR.CmpltQty          ELSE 0      END AS [Birim Maliyet] FROM     OWOR LEFT JOIN (     SELECT          IGN1.BaseRef AS İşEmriNo,         SUM(IGN1.Quantity * IGN1.Price) AS ToplamMaliyet     FROM          IGN1     INNER JOIN OIGN ON OIGN.DocEntry = IGN1.DocEntry     WHERE          IGN1.BaseType = 202     GROUP BY          IGN1.BaseRef ) AS Maliyet ON Maliyet.İşEmriNo = OWOR.DocNum WHERE     OWOR.CmpltQty > 0     AND OWOR.Status <> 'C' ORDER BY      OWOR.PostDate DESC
