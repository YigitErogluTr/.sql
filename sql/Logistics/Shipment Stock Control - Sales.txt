SELECT      T0.DocDueDate,     T0.TaxDate,     T0.DocEntry AS [Belge No],     T0.CardName,     T1.ItemCode AS [Ürün Kodu],     T2.ItemName AS [Ürün Adı],     T1.WhsCode AS [Depo Kodu],     T1.Quantity AS [Sipariş Miktarı],     T3.OnHand AS [Depodaki Mevcut Miktar],     T3.OnHand - T1.Quantity AS [Kullanılabilir Stok Miktarı],     CASE         WHEN T3.OnHand - T1.Quantity >= 0 THEN 'YETERLİ STOK'         ELSE 'YETERSİZ STOK'     END AS [Stok Durumu] FROM      ORDR T0 INNER JOIN      RDR1 T1 ON T0.DocEntry = T1.DocEntry INNER JOIN      OITM T2 ON T1.ItemCode = T2.ItemCode INNER JOIN      OITW T3 ON T1.ItemCode = T3.ItemCode AND T1.WhsCode = T3.WhsCode WHERE      T0.DocStatus = 'O'     AND T2.InvntItem = 'Y'   ORDER BY      T0.DocEntry
