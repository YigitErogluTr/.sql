SELECT       -- Talep Bilgileri     OPRQ.DocNum AS [Talep No],     OPRQ.CreateDate AS [Talep Sistem Kayıt Tarihi],     OPRQ.DocDate AS [Talep Tarihi],     OPRQ.ReqDate AS [Talep Vade Tarihi],     OPRQ.ReqName AS [Talep Eden Kullanıcı],     PRQ1.ItemCode AS [Talep Stok Kodu],     PRQ1.Dscription AS [Talep Stok Adı],     PRQ1.Quantity AS [Talep Miktarı],     PRQ1.Price AS [Talep Birim Fiyatı],     PRQ1.Currency AS [Talep Para Birimi],     PRQ1.WhsCode AS [Talep Deposu],     PRQ1.Text AS [Talep Satır Açıklaması],     OPRQ.Comments AS [Talep Açıklaması],     CASE OPRQ.DocStatus WHEN 'O' THEN 'Açık' WHEN 'C' THEN 'Kapalı' END AS [Talep Belge Durumu],     CASE OPRQ.CANCELED WHEN 'Y' THEN 'İptal Edildi' ELSE '' END AS [Talep İptal Durumu],      -- Teklif Bilgileri     OPQT.DocNum AS [Teklif No],     OPQT.CreateDate AS [Teklif Sistem Kayıt Tarihi],     OPQT.DocDate AS [Teklif Tarihi],     OPQT.DocDueDate AS [Teklif Geçerlilik Tarihi],     PQT1.ItemCode AS [Teklif Stok Kodu],     PQT1.Dscription AS [Teklif Stok Adı],     PQT1.Quantity AS [Teklif Miktarı],     PQT1.Price AS [Teklif Birim Fiyatı],     PQT1.WhsCode AS [Teklif Deposu],     PQT1.Currency AS [Teklif Para Birimi],     OPQT.Comments AS [Teklif Açıklaması],     CASE OPQT.DocStatus WHEN 'O' THEN 'Açık' WHEN 'C' THEN 'Kapalı' END AS [Teklif Belge Durumu],     CASE OPQT.CANCELED WHEN 'Y' THEN 'İptal Edildi' ELSE '' END AS [Teklif İptal Durumu],     -- Sipariş Bilgileri     OPOR.DocNum AS [Sipariş No],     OPOR.CreateDate AS [Sipariş Sistem Kayıt Tarihi],     OPOR.DocDate AS [Sipariş Tarihi],     OPOR.DocDueDate AS [Planlanan Teslim Tarihi],     POR1.Quantity AS [Sipariş Miktarı],     POR1.Price AS [Sipariş Birim Fiyatı],     POR1.Currency AS [Sipariş Para Birimi],     POR1.WhsCode AS [Sipariş Deposu],     OPOR.Comments AS [Sipariş Açıklaması],     CASE OPOR.DocStatus WHEN 'O' THEN 'Açık' WHEN 'C' THEN 'Kapalı' END AS [Sipariş Belge Durumu],     CASE OPOR.CANCELED WHEN 'Y' THEN 'İptal Edildi' ELSE '' END AS [Sipariş İptal Durumu],      -- Mal Kabul     OPDN.DocNum AS [Mal Kabul No],     OPDN.CreateDate AS [Mal Kabul Sistem Kayıt Tarihi],     OPDN.TaxDate AS [Gerçek Mal Kabul Tarihi],     PDN1.Quantity AS [Mal Kabul Miktarı],     PDN1.Price AS [Mal Kabul Birim Fiyatı],     PDN1.Currency AS [Mal Kabul Para Birimi],     PDN1.WhsCode AS [Mal Kabul Deposu],     OPDN.U_H1_eIrsaliyeNo AS [İrsaliye No],     CASE OPDN.DocStatus WHEN 'O' THEN 'Açık' WHEN 'C' THEN 'Kapalı' END AS [Mal Kabul Belge Durumu],     CASE OPDN.CANCELED WHEN 'Y' THEN 'İptal Edildi' ELSE '' END AS [Mal Kabul İptal Durumu],      -- Fatura     OPCH.DocNum AS [Fatura Belge No],     OPCH.CreateDate AS [Fatura Sistem Kayıt Tarihi],     OPCH.TaxDate AS [Fatura Tarihi],     OPCH.DocDueDate AS [Ödeme Günü],     PCH1.Quantity AS [Fatura Miktarı],     PCH1.Price AS [Fatura Birim Fiyatı],     PCH1.Currency AS [Fatura Para Birimi],     OPCH.NumAtCard AS [NumAtCard],     CASE OPCH.DocStatus WHEN 'O' THEN 'Açık' WHEN 'C' THEN 'Kapalı' END AS [Fatura Belge Durumu],     CASE OPCH.CANCELED WHEN 'Y' THEN 'İptal Edildi' ELSE '' END AS [Fatura İptal Durumu],      -- Stok Bilgileri     OITB.ItmsGrpNam AS [Kalem Grubu],     OITM.U_stokkodgrup AS [Ürün Grubu],     POR1.ItemCode AS [Stok Kodu],     POR1.Dscription AS [Stok Adı],     POR1.UomCode AS [Kalem Ölçü Birimi],      -- Tedarikçi ve Satınalmacı     OPOR.CardCode AS [Tedarikçi Kodu],     OPOR.CardName AS [Tedarikçi Adı],     OSLP.SlpName AS [Satınalmacı]  FROM      OPOR INNER JOIN POR1 ON OPOR.DocEntry = POR1.DocEntry  -- Teklif Bağlantısı (Satınalma Teklifi) LEFT JOIN PQT1 ON      POR1.BaseType = 540000006 AND     POR1.BaseEntry = PQT1.DocEntry AND     POR1.BaseLine = PQT1.LineNum  LEFT JOIN OPQT ON PQT1.DocEntry = OPQT.DocEntry  -- Talep Bağlantısı: Siparişten ya da Tekliften gelen talepler desteklenir LEFT JOIN PRQ1 ON  (     (POR1.BaseType = 1470000113 AND POR1.BaseEntry = PRQ1.DocEntry AND POR1.BaseLine = PRQ1.LineNum)     OR     (PQT1.BaseType = 1470000113 AND PQT1.BaseEntry = PRQ1.DocEntry AND PQT1.BaseLine = PRQ1.LineNum) )  LEFT JOIN OPRQ ON PRQ1.DocEntry = OPRQ.DocEntry  -- Mal Kabul Bağlantısı LEFT JOIN PDN1 ON      PDN1.BaseType = 22 AND     PDN1.BaseEntry = POR1.DocEntry AND     PDN1.BaseLine = POR1.LineNum  LEFT JOIN OPDN ON OPDN.DocEntry = PDN1.DocEntry  -- Fatura Bağlantısı LEFT JOIN PCH1 ON      PCH1.BaseType = 20 AND     PCH1.BaseEntry = PDN1.DocEntry AND     PCH1.BaseLine = PDN1.LineNum  LEFT JOIN OPCH ON OPCH.DocEntry = PCH1.DocEntry  -- Stok Bilgileri LEFT JOIN OITM ON POR1.ItemCode = OITM.ItemCode LEFT JOIN OITB ON OITM.ItmsGrpCod = OITB.ItmsGrpCod  -- Satınalmacı Bilgisi LEFT JOIN OSLP ON OPOR.SlpCode = OSLP.SlpCode  ORDER BY OPOR.DocNum, OPDN.DocNum, OPCH.DocNum, OPQT.DocNum
