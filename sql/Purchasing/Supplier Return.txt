SELECT     ORPD.DocEntry,     RPD1.ItemCode AS [Ürün Kodu],     RPD1.Dscription AS [Ürün Adı],     ORPD.NumAtCard,     ORPD.U_H1_eIrsaliyeNo,     ORPD.CardName AS [Tedarikçi Adı],     RPD1.Quantity AS [İade Miktarı],     RPD1.UomCode AS [Birim],     RPD1.Price AS [İade Birim Fiyatı],     RPD1.Currency AS [Para Birimi],     RPD1.Rate AS [Kur],          CAST(         RPD1.Quantity * RPD1.Price *          CASE              WHEN RPD1.Currency = 'TRY' OR ISNULL(RPD1.Rate, 0) = 0 THEN 1              ELSE RPD1.Rate          END          AS DECIMAL(18,2)     ) AS [Toplam Tutar (TL)],      ORPD.DocDate AS [İade Tarihi],     RPD1.InvQty as 'Stok Ölçü Birim Miktarı',     RPD1.UomCode2 as 'Stok Ölçü Birimi',     CASE          WHEN RPD1.BaseType = 20 THEN 'Satıcı Faturası'         WHEN RPD1.BaseType = 234000032 THEN 'İade Talebi'         ELSE CAST(RPD1.BaseType AS VARCHAR)     END AS [Temel Belge Türü],      RPD1.BaseEntry AS [Temel Belge No],          U1.U_NAME AS [İadeyi Oluşturan Kullanıcı]  FROM      ORPD INNER JOIN      RPD1 ON ORPD.DocEntry = RPD1.DocEntry LEFT JOIN      OUSR U1 ON ORPD.UserSign = U1.USERID WHERE      ORPD.CANCELED = 'N' ORDER BY      ORPD.DocDate DESC, ORPD.DocNum
