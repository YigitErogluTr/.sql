SELECT      T0.[DocEntry] AS "Mal Kabul Belge No",      T0.[DocDate],      T0.[CardCode],      T0.[CardName],      T1.[ItemCode],     T1.[Dscription],      T1.[Quantity],      T1.[UnitMsr],     T1.[WhsCode],      T1.[BaseRef] AS "Satınalma Siparişi",     R1.[DocEntry] AS "Satınalma Siparişi",     R1.[OpenQty] AS "Açık Miktar" ,     T2.[USER_CODE] AS "Mal Kabul Personel",      T2.[U_NAME] AS "Mal Kabul Personel Adı",      T3.[SlpName] AS "Satınalmacı" FROM OPDN T0   INNER JOIN PDN1 T1 ON T0.[DocEntry] = T1.[DocEntry]  INNER JOIN OUSR T2 ON T0.[UserSign] = T2.[USERID]  INNER JOIN OSLP T3 ON T0.[SlpCode] = T3.[SlpCode]  LEFT JOIN RDR1 R1 ON R1.[DocEntry] = T1.[BaseRef]  WHERE (T1.[WhsCode] = 'OKN' OR T1.[WhsCode] = 'IRMAKANT')  AND T0.[Canceled] <> 'Y'
