SELECT      T0.CardCode AS [Tedarikçi Kodu],     T0.CardName AS [Tedarikçi Adı],     AVG(DATEDIFF(DAY, T0.DocDueDate, T1.DocDate)) AS [Ortalama Teslim Süresi (Gün)],     SUM(T1.Quantity) AS [Toplam Teslim Edilen Miktar],     COUNT(T1.DocEntry) AS [Toplam Sipariş] FROM      OPOR T0 INNER JOIN      POR1 T1 ON T0.DocEntry = T1.DocEntry WHERE      T0.CANCELED = 'N' GROUP BY      T0.CardCode, T0.CardName ORDER BY      [Ortalama Teslim Süresi (Gün)] ASC
