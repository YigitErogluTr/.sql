SELECT     MONTH(DATEADD(DAY, ISNULL(OCTG.ExtraDays, 0), T0.DocDueDate)) AS [Ay],     YEAR(DATEADD(DAY, ISNULL(OCTG.ExtraDays, 0), T0.DocDueDate)) AS [Yıl],     SUM(T0.DocTotal) AS [Tahmini Nakit Girişi] FROM     ORDR T0 LEFT JOIN     OCTG ON T0.GroupNum = OCTG.GroupNum -- Ödeme koşullarıyla ilişkilendirme WHERE     T0.DocStatus = 'O' -- Açık siparişler GROUP BY     YEAR(DATEADD(DAY, ISNULL(OCTG.ExtraDays, 0), T0.DocDueDate)),     MONTH(DATEADD(DAY, ISNULL(OCTG.ExtraDays, 0), T0.DocDueDate)) ORDER BY     YEAR(DATEADD(DAY, ISNULL(OCTG.ExtraDays, 0), T0.DocDueDate)),     MONTH(DATEADD(DAY, ISNULL(OCTG.ExtraDays, 0), T0.DocDueDate))
