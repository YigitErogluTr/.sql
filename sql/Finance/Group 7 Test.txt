SELECT     YEAR(OJDT.RefDate) AS [Yıl],     MONTH(OJDT.RefDate) AS [Ay],      SUM(CASE WHEN LEFT(JDT1.Account, 3) = '710' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [710 - Direkt İlk Madde Malzeme],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '720' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [720 - Direkt İşçilik],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '730' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [730 - Genel Üretim Giderleri],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '740' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [740 - Hizmet Üretim Maliyeti],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '750' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [750 - Pazarlama Giderleri],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '760' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [760 - Genel Yönetim Giderleri],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '770' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [770 - Amortisman ve Finansman],     SUM(CASE WHEN LEFT(JDT1.Account, 3) = '780' THEN JDT1.Debit - JDT1.Credit ELSE 0 END) AS [780 - Finansman Giderleri]  FROM     OJDT INNER JOIN     JDT1 ON OJDT.TransId = JDT1.TransId WHERE     YEAR(OJDT.RefDate) = 2025 AND     LEFT(JDT1.Account, 3) IN ('710','720','730','740','750','760','770','780') GROUP BY     YEAR(OJDT.RefDate),     MONTH(OJDT.RefDate) ORDER BY     YEAR(OJDT.RefDate),     MONTH(OJDT.RefDate)
